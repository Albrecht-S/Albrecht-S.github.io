80 // access to core fonts:
81 // This class provides a "smart pointer" that returns a pointer to an XFontStruct.
82 // The global variable fl_xfont can be called wherever a bitmap "core" font is
83 // needed, e.g. when rendering to a GL context under X11.
84 // With Xlib / X11 fonts, fl_xfont will return the current selected font.
85 // With XFT / X11 fonts, fl_xfont will attempt to return the bitmap "core" font most
86 // similar to (usually the same as) the current XFT font.
87 class FL_EXPORT Fl_XFont_On_Demand
88 {
89 public:
90  Fl_XFont_On_Demand(XFontStruct* p = NULL) : ptr(p) { }
91  Fl_XFont_On_Demand& operator=(const Fl_XFont_On_Demand& x)
92  { ptr = x.ptr; return *this; }
93  Fl_XFont_On_Demand& operator=(XFontStruct* p)
94  { ptr = p; return *this; }
95  XFontStruct* value();
96  operator XFontStruct*() { return value(); }
97  XFontStruct& operator*() { return *value(); }
98  XFontStruct* operator->() { return value(); }
99  bool operator==(const Fl_XFont_On_Demand& x) { return ptr == x.ptr; }
100  bool operator!=(const Fl_XFont_On_Demand& x) { return ptr != x.ptr; }
101 private:
102  XFontStruct *ptr;
103 };
